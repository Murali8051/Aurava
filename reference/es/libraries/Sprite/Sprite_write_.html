Clase

Sprite
Nombre

write()
Ejemplos 
#include "Binary.h"
#include "Sprite.h"
#include "Matrix.h"

Matrix myMatrix = Matrix(0, 2, 1);

/* crea una nueva instancia Sprite
   8 pixels de ancho y 4 pixels de alto
*/
Sprite wave = Sprite(
  8, 4,
  b00011000,
  b00100100,
  b01000010,
  b10000001
);

void setup()
{ 
}

int x = 0;

void loop()
{
  myMatrix.write(x, 2, wave);     // ubica la animaci&oacute;n en la pantalla
  myMatrix.write(x - 8, 2, wave); // ubica la animaci&oacute;n de nuevo, en algun lugar de la pantalla
  delay(75);                      // espera un poco
  myMatrix.clear();               // limpia la pantalla para el el siguiente frame de la animaci&oacute;n
  if (x == 8)                      // si llega al final de la animaci&oacute;n
  {
    x = 0;                        // Empieza desde el principio
  }
  x++;                            // adelanta la coordenada x a la derecha

  // invierte los valores de todos los pixels guardados en la animaci&oacute;n 
  for (int j = 0; j  wave.height(); j++) {
    for (int i = 0; i  wave.width(); i++) {
      if (wave.read(i, j) == 1) {
        wave.write(i, j, 0);      // asigna los pixels de la animaci&oacute;n localizados en i, j como 0
      } else {
        wave.write(i, j, 1);
      }
    }
  }
}
Descripción El método write(x, y, valor) es usado para asignar los pixeles de una aminación como 1 o 0. La ubicación de los pixels es especificada en las coordenadas x, y y el estado de un pixel puede se 1 o 0.
Sintaxis 
sprite.write(x, y, valor)
Parámetros
x La posición en x
y La posición en y
valor El valor para el pixel seleccionado, puede ser 1 o 0
sprite Un objeto Sprite.
Retorna Ninguno
